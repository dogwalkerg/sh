#!/bin/bash sh_download() { country=$(curl -s ipinfo.io/country) ipv6_address=$(curl -s --max-time 1 ipv6.ip.sb) if [ "$country" = "CN" ]; then cd ~ curl -sS -O https://gh.kejilion.pro/https://raw.githubusercontent.com/kejilion/sh/main/kejilion.sh && chmod +x kejilion.sh sed -i 's/canshu="default"/canshu="CN"/g' ./kejilion.sh > /dev/null 2>&1 ./kejilion.sh elif [ -n "$ipv6_address" ]; then cd ~ curl -sS -O https://gh.kejilion.pro/https://raw.githubusercontent.com/kejilion/sh/main/kejilion.sh && chmod +x kejilion.sh sed -i 's/canshu="default"/canshu="V6"/g' ./kejilion.sh > /dev/null 2>&1 ./kejilion.sh else cd ~ curl -sS -O https://raw.githubusercontent.com/kejilion/sh/main/kejilion.sh && chmod +x kejilion.sh && ./kejilion.sh fi } sh_download
